# creating a custom function to determine the chances of rain
def RAIN_IMPTATION(cols):
    RAIN=cols[0]
    if pd.isnull(RAIN):
        return 'False'
    else:
        return RAIN
In [138]:
# custom function to determine the amount of PRCP
def PRCP_INSERT(cols):
    PRCP=cols[0]
    if pd.isnull(PRCP):
        return seattle['PRCP'].mean()
    else:
        return PRCP
